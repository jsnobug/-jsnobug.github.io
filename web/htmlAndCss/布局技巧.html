# 布局技巧

# 水平居中

# 行内元素

dom {
	text-align: center;
}

# 已知宽度的块级元素

// 方法一
dom {
	margin: 0 auto;
}

// 方法二
// 父级元素 position: relative;
dom {
	margin-left: width/2
}

# 未知宽度的块级元素

// 方法一：table标签配合margin左右auto实现水平居中。使用table标签（或直接将块级元素设值为display:table）
dom {
	display: table;
	margin: 0 atuo;
}

// 方法二
dom {
	display: inline-block;
	text-align: center;
}

// 方法三，flex布局，推荐
dom {
	display: flex;
	just-content: center;
}

// 方法四,绝对定位，推荐
dom {
	position: absolute;
	transform: translateX(-50%);
}

# 垂直居中

// 方法一：利用line-height实现，适合文字类,已知高度
dom {
	line-height: xx;
}

// 方法二,绝对定位，推荐
dom {
	position: absolute;
	transform: translateY(-50%);
}

// 方法三，flex布局，推荐
dom {
	display: flex;
	align-item: center;
}

// 方法四：table布局
// 父级转table display: table;
// 子级
dom {
	vertical-align: middle；
}
